     this.execute() 

    Option name Type Description
    context object

    Used to keep the context as the lib object instead of window when doing recursion
    start datetime

    Starting time of the test code execution
    return object

    The lib object, used for chaining

    Executes the test code, after wrapping it up on the tntlib handler code. In other words, makes the magic happen

    this.execute = function (context, start) {
    'use strict';

    var _this = context || this,
        startTime = start || new Date().getTime(),
        runTime;

    if (typeof jQuery == "function") {
        if (typeof $ !== "function") {
            window.$ = jQuery;
        }

        this.loadPlugins();

        try {
            this.writeHeaders();
            this.extendMethods();

            runTime = new Date().getTime() - startTime;
            this.runtime.total += runTime;
            this.runtime.details.push('test start >> ' + runTime + 'ms');

            this.init();

            if (typeof console != 'undefined' && console.info) {
                console.info('%cRunning campaign %c#' + this.campaign.number + '%c, recipe %c' + this.campaign.recipe.id + ' %con mbox %c' + this.campaign.mbox, this.styles.normal, this.styles.bold, this.styles.normal, this.styles.bold, this.styles.normal, this.styles.bold);
            }
        } catch (e) {
            this.bailOut(e);
        }
    } else {
        setTimeout(function () { _this.execute(_this, startTime); }, 1000);
    }

    return _this;
};

    this.extendMethods = function () {
    var _this = this;

    (function (send) {
        XMLHttpRequest.prototype.send = function (data) {
            var stateChange,
                self = this;

            function onReadyStateChange() {
                if (self.readyState == 4) {
                    for (var i = _this.ajaxQueue.length; i--;) {
                        _this.ajaxQueue[i](_this, this.responseURL);
                    }
                }

                if (stateChange) {
                    stateChange();
                }
            }

            if (this.addEventListener) {
                this.addEventListener("readystatechange", onReadyStateChange, false);
            } else {
                stateChange = this.onreadystatechange;
                this.onreadystatechange = onReadyStateChange;
            }

            send.call(this, data);
        };
    })(XMLHttpRequest.prototype.send);

    this.addRef = function (linkStr, ref) {
    'use strict';

    var link = document.createElement('a'), host, queries;

    link.href = linkStr;
    queries = link.href.match(/\?[^#?]*/g);

    if (queries) {
        link.search = "?";
        for (var i = 0; i < queries.length; i++) {
            link.search += queries[i].substring(1) + ((i < queries.length - 1) ? "&" : "");
        }
        if (link.hash) {
            link.hash = link.hash.replace(/\?[^#?]*/, "");
        }
    }

    if (link.href.indexOf("ref=") > -1) {
        return link.href.replace("ref=", "ref=" + ref + "|");
    }

    if (link.search) {
        return link.href.replace("?", "?ref=" + ref + "&");
    }

    if (link.href.indexOf("?") > -1) {
        var badQS = link.href.match(/\?([^#]*)/);
        link.href = link.href.replace(badQS[0], "");
        if (link.pathname.charAt(0) !== "/") { //IE wierdness
            host = (link.host.substr(-3) === ":80") ? link.hostname : link.host;
            return link.protocol + "//" + host + "/" + link.pathname + "?ref=" + ref + "&" + badQS[1] + link.hash;
        } else {
            return link.protocol + "//" + link.host + link.pathname + "?ref=" + ref + "&" + badQS[1] + link.hash;
        }
    }

    if (link.pathname.charAt(0) !== "/") { //IE wierdness
        host = (link.host.substr(-3) === ":80") ? link.hostname : link.host;
        return link.protocol + "//" + host + "/" + link.pathname + "?ref=" + ref + link.hash;
    } else {
        return link.protocol + "//" + link.host + link.pathname + "?ref=" + ref + link.hash;
    }

    this.wrapTimer = function (name, fn) {

        fnReturn = fn.apply(_this, args);

        runTime = new Date().getTime() - startTime;
        this.runtime.total += runTime;
        this.runtime.details.push(name + '(' + JSON.stringify(args).slice(1, -1) + ') >> ' + runTime + 'ms');

        return fnReturn;
    };
};

    this.checkNested = function (obj) {
    'use strict';
    var args = Array.prototype.slice.call(arguments);
    obj = args.shift();

    if (typeof obj === "undefined") {
        return false;
    } else {
        for (var i = 0; i < args.length; i++) {
            if (!obj.hasOwnProperty(args[i])) {
                return false;
            }
            obj = obj[args[i]];

        return true;
    }

    "use strict";

    var _this = context || this,
        startTime = start || new Date().getTime(),
        runTime;

    if ($(selector).length) {
        callback.apply($(selector), arguments);

        runTime = new Date().getTime() - startTime;
        this.runtime.total += runTime;
        this.runtime.details.push('waitForElement(' + selector + ') >> ' + runTime + 'ms');

        return;
    }

    _this.setTimeout(function () {
        _this.waitForElement(selector, callback, _this);
    }, 250, 'waitForElement "' + selector + '" callback error!');
